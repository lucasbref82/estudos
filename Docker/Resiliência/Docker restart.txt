docker container run -d -p 8080:3000 --restart=on-failure:3 lucasbref82/simulador-do-caos # Tenta reiniciar a aplicação 3 vezes caso o status seja finalizado com erro (Exited 0).
docker container run -d -p 8080:3000 --restart=unless-stopped lucasbref82/simulador-do-caos # A opção unless-stopped instrui o Docker a reiniciar o container sempre que ele parar, exceto se ele tiver sido parado manualmente pelo usuário.
docker container run -d -p 8080:3000 --restart=always lucasbref82/simulador-do-caos # A opção --restart=always faz com que o Docker sempre reinicie o container, independentemente do motivo pelo qual ele foi parado.
